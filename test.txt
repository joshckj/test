if ROOT_DIR not in sys.path:
    sys.path.insert(0, ROOT_DIR)

from sqlalchemy import create_engine
from azure.storage.blob import BlobServiceClient
from app.core.config import get_settings

settings = get_settings()

# ── PostgreSQL ─────────────────────────────────────────────────────────────────
engine = create_engine(settings.db_url)

# ── Azure Blob ─────────────────────────────────────────────────────────────────
blob_service_client = BlobServiceClient.from_connection_string(
    settings.azure_storage_connection_string
)

# ── Static folder — temporary runtime storage ──────────────────────────────────
STATIC_DIR = os.path.join(os.path.dirname(__file__), "static")
os.makedirs(STATIC_DIR, exist_ok=True)

MODEL_RUNTIME_PATH = os.path.join(STATIC_DIR, "gasleakmodel.cbm")


def download_model():
    """Pull model from Azure Blob into db/static/ for runtime use."""
    container_client = blob_service_client.get_container_client(
        settings.azure_container_name
    )
    blob_client = container_client.get_blob_client(settings.azure_model_blob_name)

    with open(MODEL_RUNTIME_PATH, "wb") as f:
        f.write(blob_client.download_blob().readall())

    print(f"[OK] Model downloaded → {MODEL_RUNTIME_PATH}")


def upload_model():
    """Push retrained model from db/static/ back to Azure Blob."""
    container_client = blob_service_client.get_container_client(
        settings.azure_container_name
    )
    blob_client = container_client.get_blob_client(settings.azure_model_blob_name)

    with open(MODEL_RUNTIME_PATH, "rb") as f:
        blob_client.upload_blob(f, overwrite=True)

    print(f"[OK] Model uploaded ← {MODEL_RUNTIME_PATH}")

# routes.py
from fastapi import APIRouter
from fastapi.responses import FileResponse, HTMLResponse
from utils.models import ChatRequest, ChatResponse
from gasleakagent.agent import run_agent
import os

router = APIRouter()

ROOT_DIR = os.path.abspath(os.path.join(os.path.dirname(__file__), "../../"))

@router.post("/chat", response_model=ChatResponse)
def chat(request: ChatRequest):
    try:
        response = run_agent(request.message)
        return ChatResponse(response=response)
    except Exception as e:
        raise HTTPException(status_code=500, detail=str(e))

@router.get("/health")
def health():
    return {"status": "ok"}

@router.get("/download/{filename}")
def download_prediction(filename: str):
    """Download a prediction CSV."""
    file_path = os.path.join(ROOT_DIR, "app", "db", "static", filename)
    if not os.path.exists(file_path):
        raise HTTPException(status_code=404, detail="File not found.")
    return FileResponse(
        path=file_path,
        filename=filename,
        media_type="text/csv"
    )

@router.get("/map/{filename}")
def view_map(filename: str):
    """Serve the Folium HTML map for the frontend to display."""
    file_path = os.path.join(ROOT_DIR, "app", "db", "static", filename)
    if not os.path.exists(file_path):
        raise HTTPException(status_code=404, detail="Map not found.")
    with open(file_path, "r") as f:
        html_content = f.read()
    return HTMLResponse(content=html_content)
